"""All github queries."""

FRAGMENTS =  """
fragment RepositoryFragment on Repository {
    id
    description
	name
}
"""
REPOSITORY_DESCRIPTION = """
query {
    resource(url: "https://github.com/mini-pw/WizualizacjaDanych2018") {
  	... RepositoryFragment
}	
"""

REPOSITORY_FORKS = """
query {
    resource(url: "https://github.com/mini-pw/WizualizacjaDanych2018") {
  		... on Repository {
			forks(first: 100, after: null) {
				nodes {
					... RepositoryFragment
				}
				pageInfo {
					endCursor
					hasNextPage
				}
			}
		}
	}
}		
"""

REPOSITORY_LANGUAGES = """
query {
    resource(url: "https://github.com/mini-pw/WizualizacjaDanych2018") {
  		... on Repository {
			languages(first:100, after: null) {
				totalCount
				nodes {
					id
					name
				}
				 pageInfo {
					endCursor
					hasNextPage
				}
			}
		}
	}
}
"""

REPOSITORY_ASIGNABLE_USERS = """
query {
    $selector {
